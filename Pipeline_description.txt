The BAGEL4 analysis strategy

# 1. HMMSEARCH: search for AOIs using hmmseach
	/usr/bagel4/bagel4_hmm_context_search.pl -dna /usr/bagel4/examples/bottromycin.fna
	
# 2. BLAST: this routine will BLAST all six frames to the 3 bacteriocin databases for AOI identification
	/usr/bagel4/bagel4_blast_context_search.pl -translate true -dna /usr/bagel4/examples/bottromycin.fna
	
# 3. COMBINE: routine to combine AOI's found by blast and primary HMMs	
	/usr/bagel4/bagel4_combine_regions.pl -s /usr/bagel4/test_plantarum -dna /var/genomes/g2d_mirror/Streptococcus_pneumoniae_R6/ASM704v1_genomic.fna
	
# 4. ANNOTATION: this routine will perform the annotation of the identified AOIs
	/usr/bagel4/bagel4_annotation_of_AOI.pl -model true -dna /usr/bagel4/test/AOI.regions.table	
	
# 5. JSON: reads the annotation file generated by bagel4_annotation_of_AOI.pl and convert it to a JSON file	
	/usr/bagel4/bagel4_gene_table_2_json.pl -s /usr/bagel4 -table gene_table_example.txt -json /var/www/ngs/anne/d3/all_genes.json
